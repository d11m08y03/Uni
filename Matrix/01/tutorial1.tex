    a + x & a - x & a - x\\
    a - x & a + x & a - x\\
    a - x & a - x & a + x\\
  \end{array} \right| = 0 \implies
  %%

    a + x & a - x & 0\\
    a - x & a + x & -2x\\
    a - x & a - x & 2x\\
  \end{array} \right| = 0 \implies
  \left| \begin{array}{ccc}
    a + x & a - x & 0\\
    a - x & a + x & -2x\\
    2(a - x) & 2a & 0\\
  \end{array} \right| = 0 \\[5pt]
  \implies
  2x [2a \cdot (a + x) - 2(a-x)^2] = 0 \implies
  2x [a \cdot (a + x) - (a-x)^2] = 0 \\ \implies 
  2x [a^2 + ax - a^2 + 2ax - x^2] = 0 \implies
  x [3ax - x^2] = 0 \implies x^2(3a - x) = 0 \\
  \therefore x = 0, 0, 3a
